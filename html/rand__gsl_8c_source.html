18/** \brief Private internal-use variable to store the random number
19 * generator.

23/** \brief Result code indicating successful completion.

25#define CAMP_RAND_GSL_SUCCESS 0

28#define CAMP_RAND_GSL_INIT_FAIL 1

36#define CAMP_RAND_GSL_ALREADY_INIT 3

38/** \brief Initialize the random number generator with the given seed.
39 *
40 * This must be called before any other GSL random number functions
41 * are called.
42 *
43 * \param seed The random seed to use.
44 * \return CAMP_RAND_GSL_SUCCESS on success, otherwise an error code.
45 * \sa camp_rand_finalize_gsl() to cleanup the generator.
46 */
47int camp_srand_gsl(int seed)
48{
49 if (camp_rand_gsl_rng) {
50 return CAMP_RAND_GSL_ALREADY_INIT;
51 }
52 gsl_set_error_handler_off(); // turn off automatic error handling
53 camp_rand_gsl_rng = gsl_rng_alloc(gsl_rng_mt19937);
54 if (camp_rand_gsl_rng == NULL) {
55 return CAMP_RAND_GSL_INIT_FAIL;
56 }
57 gsl_rng_set(camp_rand_gsl_rng, seed);
58 return CAMP_RAND_GSL_SUCCESS;
59}
60
61/** \brief Cleanup and deallocate the random number generator.
62 *
63 * This must be called after camp_srand_gsl().
64 *
65 * \return CAMP_RAND_GSL_SUCCESS on success, otherwise an error code.
66 */
67int camp_rand_finalize_gsl()
68{
69 if (!camp_rand_gsl_rng) {
70 return CAMP_RAND_GSL_NOT_INIT;
71 }
72 gsl_rng_free(camp_rand_gsl_rng);
73 camp_rand_gsl_rng = NULL;
74 return CAMP_RAND_GSL_SUCCESS;
75}

77/** \brief Generate a uniform random number in \f$[0,1)\f$.
78 *
79 * \param harvest A pointer to the generated random number.
80 * \return CAMP_RAND_GSL_SUCCESS on success, otherwise an error code.
81 */
82int camp_rand_gsl(double *harvest)
83{
84 if (!camp_rand_gsl_rng) {
85 return CAMP_RAND_GSL_NOT_INIT;
86 }
87 *harvest = gsl_rng_uniform(camp_rand_gsl_rng);
88 return CAMP_RAND_GSL_SUCCESS;
89}

91/** \brief Generate a uniform random integer in \f$[1,n]\f$.
92 *
93 * \param n The upper limit of the random integer.
94 * \param harvest A pointer to the generated random number.
95 * \return CAMP_RAND_GSL_SUCCESS on success, otherwise an error code.
96 */
97int camp_rand_int_gsl(int n, int *harvest)
98{
99 if (!camp_rand_gsl_rng) {
100 return CAMP_RAND_GSL_NOT_INIT;
101 }
102 *harvest = gsl_rng_uniform_int(camp_rand_gsl_rng, n) + 1;
103 return CAMP_RAND_GSL_SUCCESS;
104}

106/** \brief Generate a normally-distributed random number.
107 *
108 * \param mean The mean of the distribution.
109 * \param stddev The standard deviation of the distribution.
110 * \param harvest A pointer to the generated random number.
111 * \return CAMP_RAND_GSL_SUCCESS on success, otherwise an error code.
112 */
113int camp_rand_normal_gsl(double mean, double stddev, double *harvest)
114{
115 if (!camp_rand_gsl_rng) {
116 return CAMP_RAND_GSL_NOT_INIT;
117 }
118 *harvest = gsl_ran_gaussian(camp_rand_gsl_rng, stddev) + mean;
119 return CAMP_RAND_GSL_SUCCESS;
120}

122/** \brief Generate a Poisson-distributed random integer.
123 *
124 * \param mean The mean of the distribution.
125 * \param harvest A pointer to the generated random number.

128int camp_rand_poisson_gsl(double mean, int *harvest)

137/** \brief Generate a Binomial-distributed random integer.
138 *
139 * \param n The sample size for the distribution.
140 * \param p The sample probability for the distribution.
141 * \param harvest A pointer to the generated random number.
142 * \return CAMP_RAND_GSL_SUCCESS on success, otherwise an error code.
143 */
144int camp_rand_binomial_gsl(int n, double p, int *harvest)
145{
146 unsigned int u;
147
148 if (!camp_rand_gsl_rng) {
149 return CAMP_RAND_GSL_NOT_INIT;
150 }
151 u = n;
152 *harvest = gsl_ran_binomial(camp_rand_gsl_rng, p, u);
153 return CAMP_RAND_GSL_SUCCESS;
154}
