364/** \brief Solver initialization
365 *
366 * Allocate and initialize solver objects
367 *
368 * \param solver_data Pointer to a SolverData object
369 * \param abs_tol Pointer to array of absolute tolerances
370 * \param rel_tol Relative integration tolerance
371 * \param max_steps Maximum number of internal integration steps
372 * \param max_conv_fails Maximum number of convergence failures
373 */
374void solver_initialize(void *solver_data, double *abs_tol, double rel_tol,
375 int max_steps, int max_conv_fails) {
376#ifdef CAMP_USE_SUNDIALS
377 SolverData *sd; // SolverData object
378 int flag; // return code from SUNDIALS functions
379 int n_dep_var; // number of dependent variables per grid cell
380 int i_dep_var; // index of dependent variables in loops
381 int n_state_var; // number of variables on the state array per
382 // grid cell
383 int n_cells; // number of cells to solve simultaneously
384 int *var_type; // state variable types
385
386 // Seed the random number generator
387 srand((unsigned int)100);
388
389 // Get a pointer to the SolverData
390 sd = (SolverData *)solver_data;
391
392 // Create a new solver object
393 sd->cvode_mem = CVodeCreate(CV_BDF
394#ifndef SUNDIALS_VERSION_MAJOR
395# error SUNDIALS_VERSION_MAJOR not defined
396#elif SUNDIALS_VERSION_MAJOR < 4
397 , CV_NEWTON
398#endif
399 );
400 check_flag_fail((void *)sd->cvode_mem, "CVodeCreate", 0);
401
402 // Get the number of total and dependent variables on the state array,
403 // and the type of each state variable. All values are per-grid-cell.
404 n_state_var = sd->model_data.n_per_cell_state_var;
405 n_dep_var = sd->model_data.n_per_cell_dep_var;
406 var_type = sd->model_data.var_type;
407 n_cells = sd->model_data.n_cells;
408
409 // Set the solver data
410 flag = CVodeSetUserData(sd->cvode_mem, sd);
411 check_flag_fail(&flag, "CVodeSetUserData", 1);
412
413 /* Call CVodeInit to initialize the integrator memory and specify the
414 * right-hand side function in y'=f(t,y), the initial time t0, and
415 * the initial dependent variable vector y. */
416 flag = CVodeInit(sd->cvode_mem, f, (realtype)0.0, sd->y);
417 check_flag_fail(&flag, "CVodeInit", 1);
418
419 // Set the relative and absolute tolerances
420 sd->abs_tol_nv = N_VNew_Serial(n_dep_var * n_cells);
421 i_dep_var = 0;
422 for (int i_cell = 0; i_cell < n_cells; ++i_cell)
423 for (int i_spec = 0; i_spec < n_state_var; ++i_spec)
424 if (var_type[i_spec] == CHEM_SPEC_VARIABLE)
425 NV_Ith_S(sd->abs_tol_nv, i_dep_var++) = (realtype)abs_tol[i_spec];
426 flag = CVodeSVtolerances(sd->cvode_mem, (realtype)rel_tol, sd->abs_tol_nv);
427 check_flag_fail(&flag, "CVodeSVtolerances", 1);
428
429 // Add a pointer in the model data to the absolute tolerances for use during
430 // solving. TODO find a better way to do this
431 sd->model_data.abs_tol = abs_tol;
432
433 // Set the maximum number of iterations
434 flag = CVodeSetMaxNumSteps(sd->cvode_mem, max_steps);
435 check_flag_fail(&flag, "CVodeSetMaxNumSteps", 1);
436
437 // Set the maximum number of convergence failures
438 flag = CVodeSetMaxConvFails(sd->cvode_mem, max_conv_fails);
439 check_flag_fail(&flag, "CVodeSetMaxConvFails", 1);
440
441 // Set the maximum number of error test failures (TODO make separate input?)
442 flag = CVodeSetMaxErrTestFails(sd->cvode_mem, max_conv_fails);
443 check_flag_fail(&flag, "CVodeSetMaxErrTestFails", 1);
444
445 // Set the maximum number of warnings about a too-small time step
446 flag = CVodeSetMaxHnilWarns(sd->cvode_mem, MAX_TIMESTEP_WARNINGS);
447 check_flag_fail(&flag, "CVodeSetMaxHnilWarns", 1);
448
449 // Get the structure of the Jacobian matrix
450 sd->J = get_jac_init(sd);
451 sd->model_data.J_init = SUNMatClone(sd->J);
452 SUNMatCopy(sd->J, sd->model_data.J_init);
453
454 // Create a Jacobian matrix for correcting negative predicted concentrations
455 // during solving
456 sd->J_guess = SUNMatClone(sd->J);
457 SUNMatCopy(sd->J, sd->J_guess);
458
459 // Create a KLU SUNLinearSolver
460 sd->ls = SUNKLU(sd->y, sd->J);
461 check_flag_fail((void *)sd->ls, "SUNKLU", 0);
462
463 // Attach the linear solver and Jacobian to the CVodeMem object
464 flag = CVDlsSetLinearSolver(sd->cvode_mem, sd->ls, sd->J);
465 check_flag_fail(&flag, "CVDlsSetLinearSolver", 1);
466
467 // Set the Jacobian function to Jac
468 flag = CVDlsSetJacFn(sd->cvode_mem, Jac);
469 check_flag_fail(&flag, "CVDlsSetJacFn", 1);
470
471#ifdef CAMP_CUSTOM_CVODE
472 // Set a function to improve guesses for y sent to the linear solver
473 flag = CVodeSetDlsGuessHelper(sd->cvode_mem, guess_helper);
474 check_flag_fail(&flag, "CVodeSetDlsGuessHelper", 1);
475#endif
476
477// Allocate Jacobian on GPU
478#ifdef CAMP_USE_GPU
479 allocate_jac_gpu(sd->model_data.n_per_cell_solver_jac_elem, n_cells);
480#endif
481
482// Set gpu rxn values
483#ifdef CAMP_USE_GPU
484 solver_set_rxn_data_gpu(&(sd->model_data));
485#endif
486
487#ifndef FAILURE_DETAIL
488 // Set a custom error handling function
489 flag = CVodeSetErrHandlerFn(sd->cvode_mem, error_handler, (void *)sd);
490 check_flag_fail(&flag, "CVodeSetErrHandlerFn", 0);
491#endif
492
493#endif
494}

534/** \brief Solve for a given timestep
535 *
536 * \param solver_data A pointer to the initialized solver data
537 * \param state A pointer to the full state array (all grid cells)
538 * \param env A pointer to the full array of environmental conditions
539 * (all grid cells)
540 * \param t_initial Initial time (s)
541 * \param t_final (s)
542 * \return Flag indicating CAMP_SOLVER_SUCCESS or CAMP_SOLVER_FAIL
543 */
544int solver_run(void *solver_data, double *state, double *env, double t_initial,
545 double t_final) {
546#ifdef CAMP_USE_SUNDIALS
547 SolverData *sd = (SolverData *)solver_data;
548 ModelData *md = &(sd->model_data);
549 int n_state_var = sd->model_data.n_per_cell_state_var;
550 int n_cells = sd->model_data.n_cells;
551 int flag;
552
553 // Update the dependent variables
554 int i_dep_var = 0;
555 for (int i_cell = 0; i_cell < n_cells; i_cell++)
556 for (int i_spec = 0; i_spec < n_state_var; i_spec++)
557 if (sd->model_data.var_type[i_spec] == CHEM_SPEC_VARIABLE) {
558 NV_Ith_S(sd->y, i_dep_var++) =
559 state[i_spec + i_cell * n_state_var] > TINY
560 ? (realtype)state[i_spec + i_cell * n_state_var]
561 : TINY;
562 } else if (md->var_type[i_spec] == CHEM_SPEC_CONSTANT) {
563 state[i_spec + i_cell * n_state_var] =
564 state[i_spec + i_cell * n_state_var] > TINY
565 ? state[i_spec + i_cell * n_state_var]
566 : TINY;
567 }
568
569 // Update model data pointers
570 sd->model_data.total_state = state;
571 sd->model_data.total_env = env;
572
573#ifdef CAMP_DEBUG
574 // Update the debug output flag in CVODES and the linear solver
575 flag = CVodeSetDebugOut(sd->cvode_mem, sd->debug_out);
576 check_flag_fail(&flag, "CVodeSetDebugOut", 1);

579#endif

585 // (This is set up to assume the environmental variables do not change during
586 // solving. This can be changed in the future if necessary.)
587 for (int i_cell = 0; i_cell < md->n_cells; ++i_cell) {
588 // Set the grid cell state pointers

590 md->grid_cell_state = &(md->total_state[i_cell * md->n_per_cell_state_var]);
591 md->grid_cell_env = &(md->total_env[i_cell * CAMP_NUM_ENV_PARAM_]);

603 }

606
607 // Reset the flag indicating a current J_guess
608 sd->curr_J_guess = false;
609
610 // Set the initial time step
611 sd->init_time_step = (t_final - t_initial) * DEFAULT_TIME_STEP;
612
613 // Check whether there is anything to solve (filters empty air masses with no
614 // emissions)
615 if (is_anything_going_on_here(sd, t_initial, t_final) == false)
616 return CAMP_SOLVER_SUCCESS;
617

619 flag = CVodeReInit(sd->cvode_mem, t_initial, sd->y);
620 check_flag_fail(&flag, "CVodeReInit", 1);

622 // Reinitialize the linear solver

625

628 check_flag_fail(&flag, "CVodeSetInitStep", 1);
629
630 // Run the solver
631 realtype t_rt = (realtype)t_initial;
632 if (!sd->no_solve) {
633 flag = CVode(sd->cvode_mem, (realtype)t_final, sd->y, &t_rt, CV_NORMAL);
634 sd->solver_flag = flag;
635#ifndef FAILURE_DETAIL

642 printf("\nLinear Solver Setup Fail: %d %ld", lastflag, lsflag);

647 printf("\nCall to f() at failed state failed with flag %d\n", flag);
648 for (int i_cell = 0; i_cell < md->n_cells; ++i_cell) {
649 printf("\n Cell: %d ", i_cell);
650 printf("temp = %le pressure = %le\n", env[i_cell * CAMP_NUM_ENV_PARAM_],
651 env[i_cell * CAMP_NUM_ENV_PARAM_ + 1]);
652 for (int i_spec = 0, i_dep_var = 0; i_spec < md->n_per_cell_state_var;
653 i_spec++)
654 if (md->var_type[i_spec] == CHEM_SPEC_VARIABLE) {
655 printf(
656 "spec %d = %le deriv = %le\n", i_spec,

659 i_dep_var++;

664 }

666#endif
667 return CAMP_SOLVER_FAIL;
668 }
669 }

672 i_dep_var = 0;
673 for (int i_cell = 0; i_cell < n_cells; i_cell++) {
674 for (int i_spec = 0; i_spec < n_state_var; i_spec++) {
675 if (md->var_type[i_spec] == CHEM_SPEC_VARIABLE) {
676 state[i_spec + i_cell * n_state_var] =
677 (double)(NV_Ith_S(sd->y, i_dep_var) > 0.0
678 ? NV_Ith_S(sd->y, i_dep_var)
679 : 0.0);
680 i_dep_var++;
681 }
682 }
683 }
684
685 // Re-run the pre-derivative calculations to update equilibrium species
686 // and apply adjustments to final state
687 sub_model_calculate(md);
688
689 return CAMP_SOLVER_SUCCESS;
690#else
691 return CAMP_SOLVER_FAIL;
692#endif

835/** \brief Compute the time derivative f(t,y)
836 *
837 * \param t Current model time (s)
838 * \param y Dependent variable array
839 * \param deriv Time derivative vector f(t,y) to calculate
840 * \param solver_data Pointer to the solver data
841 * \return Status code
842 */
843int f(realtype t, N_Vector y, N_Vector deriv, void *solver_data) {
844 SolverData *sd = (SolverData *)solver_data;
845 ModelData *md = &(sd->model_data);
846 realtype time_step;
847
848#ifdef CAMP_DEBUG
849 sd->counterDeriv++;
850#endif
851
852 // Get a pointer to the derivative data
853 double *deriv_data = N_VGetArrayPointer(deriv);
854
855 // Get a pointer to the Jacobian estimated derivative data
856 double *jac_deriv_data = N_VGetArrayPointer(md->J_tmp);
857
858 // Get the grid cell dimensions
859 int n_cells = md->n_cells;
860 int n_state_var = md->n_per_cell_state_var;
861 int n_dep_var = md->n_per_cell_dep_var;
862
863 // Get the current integrator time step (s)
864 CVodeGetCurrentStep(sd->cvode_mem, &time_step);
865
866 // On the first call to f(), the time step hasn't been set yet, so use the
867 // default value
868 time_step = time_step > ZERO ? time_step : sd->init_time_step;
869
870 // Update the state array with the current dependent variable values.
871 // Signal a recoverable error (positive return value) for negative
872 // concentrations.
873 if (camp_solver_update_model_state(y, md, -SMALL, TINY) != CAMP_SOLVER_SUCCESS)
874 return 1;
875
876 // Get the Jacobian-estimated derivative
877 N_VLinearSum(1.0, y, -1.0, md->J_state, md->J_tmp);
878 SUNMatMatvec(md->J_solver, md->J_tmp, md->J_tmp2);
879 N_VLinearSum(1.0, md->J_deriv, 1.0, md->J_tmp2, md->J_tmp);
880
881#ifdef CAMP_DEBUG
882 // Measure calc_deriv time execution
883 clock_t start = clock();
884#endif
885
886#ifdef CAMP_USE_GPU
887 // Reset the derivative vector
888 N_VConst(ZERO, deriv);
889
890 // Calculate the time derivative f(t,y)
891 // (this is for all grid cells at once)
892 rxn_calc_deriv_gpu(md, deriv, (double)time_step);
893#endif
894
895#ifdef CAMP_DEBUG
896 clock_t end = clock();
897 sd->timeDeriv += (end - start);
898#endif
899
900 // Loop through the grid cells and update the derivative array
901 for (int i_cell = 0; i_cell < n_cells; ++i_cell) {
902 // Set the grid cell state pointers
903 md->grid_cell_id = i_cell;
904 md->grid_cell_state = &(md->total_state[i_cell * n_state_var]);
905 md->grid_cell_env = &(md->total_env[i_cell * CAMP_NUM_ENV_PARAM_]);
906 md->grid_cell_rxn_env_data =
907 &(md->rxn_env_data[i_cell * md->n_rxn_env_data]);
908 md->grid_cell_aero_rep_env_data =
909 &(md->aero_rep_env_data[i_cell * md->n_aero_rep_env_data]);
910 md->grid_cell_sub_model_env_data =
911 &(md->sub_model_env_data[i_cell * md->n_sub_model_env_data]);
912
913 // Update the aerosol representations
914 aero_rep_update_state(md);
915
916 // Run the sub models
917 sub_model_calculate(md);
918
919#ifdef CAMP_DEBUG
920 // Measure calc_deriv time execution
921 clock_t start2 = clock();
922#endif
923
924#ifndef CAMP_USE_GPU
925 // Reset the TimeDerivative
926 time_derivative_reset(sd->time_deriv);
927
928 // Calculate the time derivative f(t,y)
929 rxn_calc_deriv(md, sd->time_deriv, (double)time_step);
930
931 // Update the deriv array
932 if (sd->use_deriv_est == 1) {
933 time_derivative_output(sd->time_deriv, deriv_data, jac_deriv_data,
934 sd->output_precision);
935 } else {
936 time_derivative_output(sd->time_deriv, deriv_data, NULL,
937 sd->output_precision);
938 }
939#else
940 // Add contributions from reactions not implemented on GPU
941 // FIXME need to fix this to use TimeDerivative
942 rxn_calc_deriv_specific_types(md, sd->time_deriv, (double)time_step);
943#endif
944
945#ifdef CAMP_DEBUG
946 clock_t end2 = clock();
947 sd->timeDeriv += (end2 - start2);
948 sd->max_loss_precision = time_derivative_max_loss_precision(sd->time_deriv);
949#endif
950
951 // Advance the derivative for the next cell
952 deriv_data += n_dep_var;
953 jac_deriv_data += n_dep_var;
954 }
955
956 // Return 0 if success
957 return (0);
958}

960/** \brief Compute the Jacobian
961 *
962 * \param t Current model time (s)
963 * \param y Dependent variable array
964 * \param deriv Time derivative vector f(t,y)
965 * \param J Jacobian to calculate
966 * \param solver_data Pointer to the solver data
967 * \param tmp1 Unused vector
968 * \param tmp2 Unused vector
969 * \param tmp3 Unused vector
970 * \return Status code
971 */

973 N_Vector tmp1, N_Vector tmp2, N_Vector tmp3) {
974 SolverData *sd = (SolverData *)solver_data;
975 ModelData *md = &(sd->model_data);
976 realtype time_step;
977
978#ifdef CAMP_DEBUG
979 sd->counterJac++;
980#endif
981
982 // Get the grid cell dimensions
983 int n_state_var = md->n_per_cell_state_var;
984 int n_dep_var = md->n_per_cell_dep_var;
985 int n_cells = md->n_cells;

988 double *J_param_data = SM_DATA_S(md->J_params);
989 double *J_rxn_data = SM_DATA_S(md->J_rxn);
990 // Initialize the sparse matrix (sized for one grid cell)
991 // solver_data->model_data.J_rxn =
992 // SUNSparseMatrix(n_state_var, n_state_var, n_jac_elem_rxn, CSC_MAT);
993
994 // TODO: use this instead of saving all this jacs
995 // double J_rxn_data[md->n_per_cell_dep_var];
996 // memset(J_rxn_data, 0, md->n_per_cell_dep_var * sizeof(double));
997
998 // double *J_rxn_data = (double*)calloc(md->n_per_cell_state_var,
999 // sizeof(double));
1000
1001 // !!!! Do not use tmp2 - it is the same as y !!!! //
1002 // FIXME Find out why cvode is sending tmp2 as y
1003
1004 // Calculate the the derivative for the current state y without
1005 // the estimated derivative from the last Jacobian calculation
1006 sd->use_deriv_est = 0;
1007 if (f(t, y, deriv, solver_data) != 0) {
1008 printf("\n Derivative calculation failed.\n");
1009 sd->use_deriv_est = 1;
1010 return 1;
1011 }
1012 sd->use_deriv_est = 1;
1013
1014 // Update the state array with the current dependent variable values
1015 // Signal a recoverable error (positive return value) for negative
1016 // concentrations.
1017 if (camp_solver_update_model_state(y, md, -SMALL, TINY) != CAMP_SOLVER_SUCCESS)
1018 return 1;
1019
1020 // Get the current integrator time step (s)
1021 CVodeGetCurrentStep(sd->cvode_mem, &time_step);
1022
1023 // Reset the primary Jacobian
1024 /// \todo #83 Figure out how to stop CVODE from resizing the Jacobian
1025 /// during solving
1026 SM_NNZ_S(J) = SM_NNZ_S(md->J_init);
1027 for (int i = 0; i <= SM_NP_S(J); i++) {
1028 (SM_INDEXPTRS_S(J))[i] = (SM_INDEXPTRS_S(md->J_init))[i];
1029 }
1030 for (int i = 0; i < SM_NNZ_S(J); i++) {
1031 (SM_INDEXVALS_S(J))[i] = (SM_INDEXVALS_S(md->J_init))[i];
1032 (SM_DATA_S(J))[i] = (realtype)0.0;
1033 }
1034
1035#ifdef CAMP_DEBUG
1036 clock_t start2 = clock();
1037#endif
1038
1039#ifdef CAMP_USE_GPU
1040 // Calculate the Jacobian
1041 rxn_calc_jac_gpu(md, J, time_step);
1042#endif
1043
1044#ifdef CAMP_DEBUG
1045 clock_t end2 = clock();
1046 sd->timeJac += (end2 - start2);
1047#endif
1048
1049 // Solving on CPU only
1050
1051 // Loop over the grid cells to calculate sub-model and rxn Jacobians
1052 for (int i_cell = 0; i_cell < n_cells; ++i_cell) {
1053 // Set the grid cell state pointers
1054 md->grid_cell_id = i_cell;
1055 md->grid_cell_state = &(md->total_state[i_cell * n_state_var]);
1056 md->grid_cell_env = &(md->total_env[i_cell * CAMP_NUM_ENV_PARAM_]);
1057 md->grid_cell_rxn_env_data =
1058 &(md->rxn_env_data[i_cell * md->n_rxn_env_data]);
1059 md->grid_cell_aero_rep_env_data =
1060 &(md->aero_rep_env_data[i_cell * md->n_aero_rep_env_data]);
1061 md->grid_cell_sub_model_env_data =
1062 &(md->sub_model_env_data[i_cell * md->n_sub_model_env_data]);
1063
1064 // Reset the sub-model and reaction Jacobians
1065 for (int i = 0; i < SM_NNZ_S(md->J_params); ++i)
1066 SM_DATA_S(md->J_params)[i] = 0.0;
1067 jacobian_reset(sd->jac);
1068
1069 // Update the aerosol representations
1070 aero_rep_update_state(md);
1071
1072 // Run the sub models and get the sub-model Jacobian
1073 sub_model_calculate(md);
1074 sub_model_get_jac_contrib(md, J_param_data, time_step);
1075 CAMP_DEBUG_JAC(md->J_params, "sub-model Jacobian");
1076
1077#ifdef CAMP_DEBUG
1078 clock_t start = clock();
1079#endif
1080
1081#ifndef CAMP_USE_GPU
1082 // Calculate the reaction Jacobian
1083 rxn_calc_jac(md, sd->jac, time_step);
1084#else
1085 // Add contributions from reactions not implemented on GPU
1086 rxn_calc_jac_specific_types(md, sd->jac, time_step);
1087#endif
1088
1089// rxn_calc_jac_specific_types(md, J_rxn_data, time_step);
1090#ifdef CAMP_DEBUG
1091 clock_t end = clock();
1092 sd->timeJac += (end - start);
1093#endif
1094
1095 // Output the Jacobian to the SUNDIALS J_rxn
1096 jacobian_output(sd->jac, SM_DATA_S(md->J_rxn));
1097 CAMP_DEBUG_JAC(md->J_rxn, "reaction Jacobian");
1098
1099 // Set the solver Jacobian using the reaction and sub-model Jacobians
1100 JacMap *jac_map = md->jac_map;
1101 SM_DATA_S(md->J_params)[0] = 1.0; // dummy value for non-sub model calcs
1102 for (int i_map = 0; i_map < md->n_mapped_values; ++i_map)
1103 SM_DATA_S(J)
1104 [i_cell * md->n_per_cell_solver_jac_elem + jac_map[i_map].solver_id] +=
1105 SM_DATA_S(md->J_rxn)[jac_map[i_map].rxn_id] *
1106 SM_DATA_S(md->J_params)[jac_map[i_map].param_id];
1107 CAMP_DEBUG_JAC(J, "solver Jacobian");
1108 }
1109
1110 // Save the Jacobian for use with derivative calculations
1111 for (int i_elem = 0; i_elem < SM_NNZ_S(J); ++i_elem)
1112 SM_DATA_S(md->J_solver)[i_elem] = SM_DATA_S(J)[i_elem];
1113 N_VScale(1.0, y, md->J_state);
1114 N_VScale(1.0, deriv, md->J_deriv);
1115
1116#ifdef CAMP_DEBUG

1128/** \brief Check a Jacobian for accuracy
1129 *
1130 * This function compares Jacobian elements against differences in derivative
1131 * calculations for small changes to the state array:
1132 * \f[
1133 * J_{ij}(x) = \frac{f_i(x+\sum_j e_j) - f_i(x)}{\epsilon}
1134 * \f]
1135 * where \f$\epsilon_j = 10^{-8} \left|x_j\right|\f$
1136 *
1137 * \param t Current time [s]
1138 * \param y Current state array
1139 * \param J Jacobian matrix to evaluate
1140 * \param deriv Current derivative \f$f(y)\f$
1141 * \param tmp Working array the size of \f$y\f$
1142 * \param tmp1 Working array the size of \f$y\f$
1143 * \param solver_data Solver data
1144 * \return True if Jacobian values are accurate, false otherwise
1145 */
1146bool check_Jac(realtype t, N_Vector y, SUNMatrix J, N_Vector deriv,
1147 N_Vector tmp, N_Vector tmp1, void *solver_data) {
1148 realtype *d_state = NV_DATA_S(y);
1149 realtype *d_deriv = NV_DATA_S(deriv);
1150 bool retval = true;
1151
1152#ifdef CAMP_USE_GSL
1153 GSLParam gsl_param;
1154 gsl_function gsl_func;
1155
1156 // Set up gsl parameters needed during numerical differentiation
1157 gsl_param.t = t;
1158 gsl_param.y = tmp;
1159 gsl_param.deriv = tmp1;
1160 gsl_param.solver_data = (SolverData *)solver_data;
1161
1162 // Set up the gsl function
1163 gsl_func.function = &gsl_f;
1164 gsl_func.params = &gsl_param;
1165#endif
1166
1167 // Calculate the the derivative for the current state y
1168 if (f(t, y, deriv, solver_data) != 0) {
1169 printf("\n Derivative calculation failed.\n");
1170 return false;
1171 }
1172
1173 // Loop through the independent variables, numerically calculating
1174 // the partial derivatives d_fy/d_x
1175 for (int i_ind = 0; i_ind < NV_LENGTH_S(y); ++i_ind) {
1176 // If GSL is available, use their numerical differentiation to
1177 // calculate the partial derivatives. Otherwise, estimate them by
1178 // advancing the state.
1179#ifdef CAMP_USE_GSL
1180
1181 // Reset tmp to the initial state
1182 N_VScale(ONE, y, tmp);
1183
1184 // Save the independent species concentration and index
1185 double x = d_state[i_ind];
1186 gsl_param.ind_var = i_ind;
1187
1188 // Skip small concentrations
1189 if (x < SMALL) continue;
1190
1191 // Do the numerical differentiation for each potentially non-zero
1192 // Jacobian element
1193 for (int i_elem = SM_INDEXPTRS_S(J)[i_ind];
1194 i_elem < SM_INDEXPTRS_S(J)[i_ind + 1]; ++i_elem) {
1195 int i_dep = SM_INDEXVALS_S(J)[i_elem];
1196
1197 double abs_err;
1198 double partial_deriv;
1199
1200 gsl_param.dep_var = i_dep;
1201
1202 bool test_pass = false;
1203 double h, abs_tol, rel_diff, scaling;
1204
1205 // Evaluate the Jacobian element over a range of initial step sizes
1206 for (scaling = JAC_CHECK_ADV_MIN;
1207 scaling <= JAC_CHECK_ADV_MAX && test_pass == false;
1208 scaling *= 10.0) {
1209 // Get the current initial step size
1210 h = x * scaling;
1211
1212 // Get the partial derivative d_fy/dx
1213 if (gsl_deriv_forward(&gsl_func, x, h, &partial_deriv, &abs_err) == 1) {
1214 printf("\nERROR in numerical differentiation for J[%d][%d]", i_ind,
1215 i_dep);
1216 }
1217
1218 // Evaluate the results
1219 abs_tol = 1.2 * fabs(abs_err);
1220 abs_tol =
1221 abs_tol > JAC_CHECK_GSL_ABS_TOL ? abs_tol : JAC_CHECK_GSL_ABS_TOL;
1222 rel_diff = 1.0;
1223 if (partial_deriv != 0.0)
1224 rel_diff =
1225 fabs((SM_DATA_S(J)[i_elem] - partial_deriv) / partial_deriv);
1226 if (fabs(SM_DATA_S(J)[i_elem] - partial_deriv) < abs_tol ||
1227 rel_diff < JAC_CHECK_GSL_REL_TOL)
1228 test_pass = true;
1229 }
1230
1231 // If the test does not pass with any initial step size, print out the
1232 // failure, output the local derivative state and return false
1233 if (test_pass == false) {
1234 printf(
1235 "\nError in Jacobian[%d][%d]: Got %le; expected %le"
1236 "\n difference %le is greater than error %le",
1237 i_ind, i_dep, SM_DATA_S(J)[i_elem], partial_deriv,
1238 fabs(SM_DATA_S(J)[i_elem] - partial_deriv), abs_tol);
1239 printf("\n relative error %le intial step size %le", rel_diff, h);
1240 printf("\n initial rate %le initial state %le", d_deriv[i_dep],
1241 d_state[i_ind]);
1242 printf(" scaling %le", scaling);
1243 ModelData *md = &(((SolverData *)solver_data)->model_data);
1244 for (int i_cell = 0; i_cell < md->n_cells; ++i_cell)
1245 for (int i_spec = 0; i_spec < md->n_per_cell_state_var; ++i_spec)
1246 printf("\n cell: %d species %d state_id %d conc: %le", i_cell,
1247 i_spec, i_cell * md->n_per_cell_state_var + i_spec,
1248 md->total_state[i_cell * md->n_per_cell_state_var + i_spec]);
1249 retval = false;
1250 output_deriv_local_state(t, y, deriv, solver_data, &f, i_dep, i_ind,
1251 SM_DATA_S(J)[i_elem], h / 10.0);
1252 }
1253 }
1254#endif
1255 }
1256 return retval;
1257}

1290/** \brief Try to improve guesses of y sent to the linear solver
1291 *
1292 * This function checks if there are any negative guessed concentrations,
1293 * and if there are it calculates a set of initial corrections to the
1294 * guessed state using the state at time \f$t_{n-1}\f$ and the derivative
1295 * \f$f_{n-1}\f$ and advancing the state according to:
1296 * \f[
1297 * y_n = y_{n-1} + \sum_{j=1}^m h_j * f_j
1298 * \f]
1299 * where \f$h_j\f$ is the largest timestep possible where
1300 * \f[
1301 * y_{j-1} + h_j * f_j > 0
1302 * \f]
1303 * and
1304 * \f[
1305 * t_n = t_{n-1} + \sum_{j=1}^m h_j
1306 * \f]
1307 *
1308 * \param t_n Current time [s]
1309 * \param h_n Current time step size [s] If this is set to zero, the change hf
1310 * is assumed to be an adjustment where y_n = y_n1 + hf
1311 * \param y_n Current guess for \f$y(t_n)\f$
1312 * \param y_n1 \f$y(t_{n-1})\f$
1313 * \param hf Current guess for change in \f$y\f$ from \f$t_{n-1}\f$ to
1314 * \f$t_n\f$ [input/output]
1315 * \param solver_data Solver data
1316 * \param tmp1 Temporary vector for calculations
1317 * \param corr Vector of calculated adjustments to \f$y(t_n)\f$ [output]
1318 * \return 1 if corrections were calculated, 0 if not
1319 */
1320#ifdef CAMP_CUSTOM_CVODE
1321int guess_helper(const realtype t_n, const realtype h_n, N_Vector y_n,
1322 N_Vector y_n1, N_Vector hf, void *solver_data, N_Vector tmp1,
1323 N_Vector corr) {
1324 SolverData *sd = (SolverData *)solver_data;
1325 realtype *ay_n = NV_DATA_S(y_n);
1326 realtype *ay_n1 = NV_DATA_S(y_n1);
1327 realtype *atmp1 = NV_DATA_S(tmp1);
1328 realtype *acorr = NV_DATA_S(corr);
1329 realtype *ahf = NV_DATA_S(hf);
1330 int n_elem = NV_LENGTH_S(y_n);
1331
1332 // Only try improvements when negative concentrations are predicted
1333 if (N_VMin(y_n) > -SMALL) return 0;
1334
1335 CAMP_DEBUG_PRINT_FULL("Trying to improve guess");
1336
1337 // Copy \f$y(t_{n-1})\f$ to working array
1338 N_VScale(ONE, y_n1, tmp1);
1339
1340 // Get \f$f(t_{n-1})\f$
1341 if (h_n > ZERO) {
1342 N_VScale(ONE / h_n, hf, corr);
1343 } else {
1344 N_VScale(ONE, hf, corr);
1345 }
1346 CAMP_DEBUG_PRINT("Got f0");
1347
1348 // Advance state interatively
1349 realtype t_0 = h_n > ZERO ? t_n - h_n : t_n - ONE;
1350 realtype t_j = ZERO;
1351 int iter = 0;
1352 for (; iter < GUESS_MAX_ITER && t_0 + t_j < t_n; iter++) {
1353 // Calculate \f$h_j\f$
1354 realtype h_j = t_n - (t_0 + t_j);
1355 int i_fast = -1;
1356 for (int i = 0; i < n_elem; i++) {
1357 realtype t_star = -atmp1[i] / acorr[i];
1358 if ((t_star > ZERO || (t_star == ZERO && acorr[i] < ZERO)) &&
1359 t_star < h_j) {
1360 h_j = t_star;
1361 i_fast = i;
1362 }
1363 }
1364
1365 // Scale incomplete jumps
1366 if (i_fast >= 0 && h_n > ZERO)
1367 h_j *= 0.95 + 0.1 * iter / (double)GUESS_MAX_ITER;
1368 h_j = t_n < t_0 + t_j + h_j ? t_n - (t_0 + t_j) : h_j;
1369
1370 // Only make small changes to adjustment vectors used in Newton iteration
1371 if (h_n == ZERO &&
1372 t_n - (h_j + t_j + t_0) > ((CVodeMem)sd->cvode_mem)->cv_reltol)
1373 return -1;
1374
1375 // Advance the state
1376 N_VLinearSum(ONE, tmp1, h_j, corr, tmp1);
1377 CAMP_DEBUG_PRINT_FULL("Advanced state");
1378
1379 // Advance t_j
1380 t_j += h_j;
1381
1382 // Recalculate the time derivative \f$f(t_j)\f$
1383 if (f(t_0 + t_j, tmp1, corr, solver_data) != 0) {
1384 CAMP_DEBUG_PRINT("Unexpected failure in guess helper!");
1385 N_VConst(ZERO, corr);
1386 return -1;
1387 }
1388 ((CVodeMem)sd->cvode_mem)->cv_nfe++;
1389
1390 if (iter == GUESS_MAX_ITER - 1 && t_0 + t_j < t_n) {
1391 CAMP_DEBUG_PRINT("Max guess iterations reached!");
1392 if (h_n == ZERO) return -1;
1393 }
1394 }
1395
1396 CAMP_DEBUG_PRINT_INT("Guessed y_h in steps:", iter);
1397
1398 // Set the correction vector
1399 N_VLinearSum(ONE, tmp1, -ONE, y_n, corr);
1400
1401 // Scale the initial corrections
1402 if (h_n > ZERO) N_VScale(0.999, corr, corr);
1403
1404 // Update the hf vector
1405 N_VLinearSum(ONE, tmp1, -ONE, y_n1, hf);
1406
1407 return 1;
1408}

1642 }

1644
1645 // Set map indices (when no sub-model value is used, the param_id is
1646 // set to 0 which maps to a fixed value of 1.0
1647 int i_mapped_value = 0;
1648 for (unsigned int i_ind = 0; i_ind < n_state_var; ++i_ind) {
1649 for (unsigned int i_elem =
1650 jacobian_column_pointer_value(solver_data->jac, i_ind);
1651 i_elem < jacobian_column_pointer_value(solver_data->jac, i_ind + 1);
1652 ++i_elem) {
1653 unsigned int i_dep = jacobian_row_index(solver_data->jac, i_elem);
1654 // skip dependent species that are not solver variables and
1655 // depenedent species that aren't used by any reaction
1656 if (solver_data->model_data.var_type[i_dep] != CHEM_SPEC_VARIABLE ||
1657 jacobian_get_element_id(solver_data->jac, i_dep, i_ind) == -1)
1658 continue;

1660 if (solver_data->model_data.var_type[i_ind] == CHEM_SPEC_VARIABLE &&
1661 solver_data->model_data.var_type[i_dep] == CHEM_SPEC_VARIABLE) {
1662 map[i_mapped_value].solver_id =
1663 jacobian_get_element_id(solver_jac, i_dep, i_ind);
1664 map[i_mapped_value].rxn_id = i_elem;
1665 map[i_mapped_value].param_id = 0;
1666 ++i_mapped_value;
1667 continue;
1668 }
1669 // Check the sub model Jacobian for remaining conditions
1670 // (variable dependent species; independent parameter from sub model)
1671 for (int j_ind = 0; j_ind < n_state_var; ++j_ind) {
1672 if (jacobian_get_element_id(param_jac, i_ind, j_ind) != -1 &&
1673 solver_data->model_data.var_type[j_ind] == CHEM_SPEC_VARIABLE) {
1674 map[i_mapped_value].solver_id =
1675 jacobian_get_element_id(solver_jac, i_dep, j_ind);
1676 map[i_mapped_value].rxn_id = i_elem;
1677 map[i_mapped_value].param_id =
1678 jacobian_get_element_id(param_jac, i_ind, j_ind);
1679 ++i_mapped_value;
1680 }
1681 }
1682 }
1683 }

1685 SolverData *sd = solver_data;
1686 CAMP_DEBUG_JAC_STRUCT(sd->model_data.J_params, "Param struct");
1687 CAMP_DEBUG_JAC_STRUCT(sd->model_data.J_rxn, "Reaction struct");
1688 CAMP_DEBUG_JAC_STRUCT(M, "Solver struct");
1689
1690 if (i_mapped_value != n_mapped_values) {
1691 printf("[ERROR-340355266] Internal error");
1692 exit(EXIT_FAILURE);
1693 }
1694
1695 // Create vectors to store Jacobian state and derivative data
1696 solver_data->model_data.J_state = N_VClone(solver_data->y);
1697 solver_data->model_data.J_deriv = N_VClone(solver_data->y);
1698 solver_data->model_data.J_tmp = N_VClone(solver_data->y);
1699 solver_data->model_data.J_tmp2 = N_VClone(solver_data->y);
1700
1701 // Initialize the Jacobian state and derivative arrays to zero
1702 // for use before the first call to Jac()
1703 N_VConst(0.0, solver_data->model_data.J_state);
1704 N_VConst(0.0, solver_data->model_data.J_deriv);
1705
1706 // Free the memory used
1707 jacobian_free(&param_jac);
1708 jacobian_free(&solver_jac);
1709 free(deriv_ids);
1710
