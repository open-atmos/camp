35/** \brief Get the Jacobian elements used by a particular reaction
36 *
37 * \param model_data A pointer to the model data
38 * \param jac Jacobian
39 */
40void rxn_get_used_jac_elem(ModelData *model_data, Jacobian *jac) {
41 // Get the number of reactions
42 int n_rxn = model_data->n_rxn;
43
44 // Loop through the reactions to determine the Jacobian elements used
45 // advancing the rxn_data pointer each time
46 for (int i_rxn = 0; i_rxn < n_rxn; i_rxn++) {
47 // Get pointers to the reaction data
48 int *rxn_int_data =
49 &(model_data->rxn_int_data[model_data->rxn_int_indices[i_rxn]]);
50 double *rxn_float_data =
51 &(model_data->rxn_float_data[model_data->rxn_float_indices[i_rxn]]);
52
53 // Get the reaction type
54 int rxn_type = *(rxn_int_data++);
55
56 // Call the appropriate function
57 switch (rxn_type) {
58 case RXN_AQUEOUS_EQUILIBRIUM:
59 rxn_aqueous_equilibrium_get_used_jac_elem(rxn_int_data, rxn_float_data,
60 jac);

64 break;

70 break;

603 * \param solver_data Pointer to solver data

610 ModelData *model_data =

613 // Get pointers to the reaction data

627 model_data->rxn_env_idx[model_data->n_added_rxns + 1] =
628 model_data->rxn_env_idx[model_data->n_added_rxns] + n_env_param;
629 ++(model_data->n_added_rxns);
630
631 // Add the reaction type
632 *(rxn_int_data++) = rxn_type;
633
634 // Add integer parameters

636
637 // Add floating-point parameters
638 for (; n_float_param > 0; --n_float_param)

644/** \brief Update reaction data
645 *
646 * Update data for one or more reactions. Reactions of a certain type are
647 * passed a void pointer to updated data that must be in the format specified
648 * by the reaction type. This data could be used to find specific reactions of
649 * the specified type and, for example, update rate constants.
650 *
651 * \param cell_id Id of the grid cell to update
652 * \param rxn_id Id of the reaction (or 0 if unknown)
653 * \param update_rxn_type Type of the reaction
654 * \param update_data Pointer to updated data to pass to the reaction
655 * \param solver_data Pointer to solver data
656 */
657void rxn_update_data(int cell_id, int *rxn_id, int update_rxn_type,
658 void *update_data, void *solver_data) {
659 ModelData *model_data =
660 (ModelData *)&(((SolverData *)solver_data)->model_data);
661
662 // Point to the environment-dependent data for the grid cell
663 model_data->grid_cell_rxn_env_data =
664 &(model_data->rxn_env_data[cell_id * model_data->n_rxn_env_data]);
665
666 // Get the number of reactions
667 int n_rxn = model_data->n_rxn;
668
669 // Loop through the reactions advancing the rxn_data pointer each time
670 for (; (*rxn_id) < n_rxn; (*rxn_id)++) {
671 // Get pointers to the reaction data
672 int *rxn_int_data =
673 &(model_data->rxn_int_data[model_data->rxn_int_indices[*rxn_id]]);
674 double *rxn_float_data =
675 &(model_data->rxn_float_data[model_data->rxn_float_indices[*rxn_id]]);
676
677 double *rxn_env_data =
678 &(model_data->grid_cell_rxn_env_data[model_data->rxn_env_idx[*rxn_id]]);
679
680 // Get the reaction type
681 int rxn_type = *(rxn_int_data++);
682
683 bool found = false;
684
685 // Try the update data function for reactions of the correct type
686 if (rxn_type == update_rxn_type) {
687 switch (rxn_type) {
688 case RXN_EMISSION:
689 found = rxn_emission_update_data((void *)update_data, rxn_int_data,
690 rxn_float_data, rxn_env_data);
691 break;
692 case RXN_FIRST_ORDER_LOSS:
693 found = rxn_first_order_loss_update_data(
694 (void *)update_data, rxn_int_data, rxn_float_data, rxn_env_data);
695 break;
696 case RXN_PHOTOLYSIS:
697 found = rxn_photolysis_update_data((void *)update_data, rxn_int_data,
698 rxn_float_data, rxn_env_data);
699 break;
700 case RXN_WET_DEPOSITION:
701 found = rxn_wet_deposition_update_data(
702 (void *)update_data, rxn_int_data, rxn_float_data, rxn_env_data);
703 break;
704 }
705 if (found) return;
706 }
707 }
