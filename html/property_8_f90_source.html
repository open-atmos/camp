138 !> Load a property set from input data
139#ifdef CAMP_USE_JSON
140 recursive subroutine load(this, json, j_obj, as_object, owner_name, &
141 allow_duplicates)
142
143 !> Property dataset
144 class(property_t), intent(inout) :: this
145 !> JSON core
146 type(json_core), pointer, intent(in) :: json
147 !> JSON object
148 type(json_value), pointer, intent(in) :: j_obj
149 !> Set to true if j_obj is a json object to parse, adding all child
150 !! key-value pairs to the data set, or false if j_obj is a single
151 !! key-value pair to add to the data set
152 logical, intent(in) :: as_object
153 !> Name of the owner of the property set. For use in error messages
154 character(len=*), intent(in):: owner_name
155 !> Flag to indicate whether to allow duplicate keys. Defaults to false
156 logical, intent(in), optional :: allow_duplicates
157
158 type(json_value), pointer :: child, next
159 type(property_t), pointer :: sub_prop
160 character(kind=json_ck, len=:), allocatable :: unicode_prop_key
161 character(len=:), allocatable :: prop_key
162 logical :: allow_dup
163
164 character(kind=json_ck, len=:), allocatable :: unicode_val
165 character(len=:), allocatable :: str_val
166 logical(json_lk) :: bool_val
167 real(json_rk) :: real_val
168 integer(json_ik) :: int_val
169
170 integer(json_ik) :: var_type
171
172 allow_dup = .false.
173
174 if (present(allow_duplicates)) allow_dup = allow_duplicates
175
176 ! initialize pointer to next object to parse
177 next => null()
178
179 ! determine whether to add parent or children key-value pairs
180 if (as_object) then
181 call json%get_child(j_obj, child)
182 else
183 child => j_obj
184 end if
185
186 ! loop through set of json objects to add to the property set
187 do while (associated(child))
188
189 ! get the key name and value
190 call json%info(child, name=unicode_prop_key, var_type=var_type)
191 prop_key = unicode_prop_key
192
193 ! add key-value pair of appropriate type
194 select case (var_type)
195
196 ! skip null objects
197 case (json_null)
198
199 ! integer
200 case (json_integer)
201 call json%get(child, int_val)
202 call this%put(prop_key, int(int_val, i_kind), allow_dup, &
203 owner_name)
204
205 ! double
206 case (json_double)
207 call json%get(child, real_val)
208 call this%put(prop_key, real(real_val, dp), allow_dup, &
209 owner_name)
210
211 ! boolean
212 case (json_logical)
213 call json%get(child, bool_val)
214 call this%put(prop_key, logical(bool_val), allow_dup, &
215 owner_name)
216
217 ! string
218 case (json_string)
219 call json%get(child, unicode_val)
220 str_val = unicode_val
221 call this%put(prop_key, str_val, allow_dup, owner_name)
222
223 ! sub-set of key-value pairs
224 case (json_object)
225 sub_prop => property_t()
226 call sub_prop%load(json, child, .true., owner_name, allow_dup)
227 call this%put(prop_key, sub_prop, allow_dup, owner_name)
228 deallocate(sub_prop)
229
230 ! sub-set of values
231 case (json_array)
232 sub_prop => property_t()
233 call sub_prop%load(json, child, .true., owner_name, allow_dup)
234 call this%put(prop_key, sub_prop, allow_dup, owner_name)
235 deallocate(sub_prop)
236
237 ! skip other types
238 case default
239 end select
240
241 ! get the next object to add
242 if (as_object) call json%get_next(child, next)
243 child => next
244
245 end do
246
247#else
248 subroutine load(this)
249
250 !> Property dataset
251 class(property_t), intent(inout) :: this
252
253 call warn_msg(733896496, "No support for input files.")
254#endif
255 end subroutine load

259 !> Put an element in the property data set
260 recursive subroutine put(this, key, val, allow_duplicates, owner_name)
261
262 !> Property data set
263 class(property_t), intent(inout) :: this
264 !> New key
265 character(len=*), intent(in) :: key
266 !> New value
267 class(*), intent(in) :: val
268 !> Flag indicating whether to allow duplicate keys
269 logical, intent(in) :: allow_duplicates
270 !> Name of owner of the property set. For use in error messages
271 character(len=*), intent(in) :: owner_name
272
273 type(property_link_t), pointer :: new_link, sub_link
274 type(property_t), allocatable :: sub_prop_set
275 class(*), pointer :: curr_val
276
277 ! if this is an array element, the key will be empty
278 if (len(key).ge.1) then
279
280 ! look for the key in the existing properties if disallowing duplictes
281 if (.not.allow_duplicates) then
282
283 new_link => this%get(key)
284
285 ! do not allow overwrites of existing properties, but allow sub-sets
286 ! of properties to be appended
287 if (associated(new_link)) then
288 curr_val => new_link%val
289 select type (curr_val)
290 class is (property_t)
291 select type (val)
292 class is (property_t)
293 sub_link => val%first_link
294 do while (associated(sub_link))
295 call curr_val%put(sub_link%key_name, sub_link%val, .false., &
296 owner_name)
297 sub_link => sub_link%next_link
298 end do
299 class default
300 call new_link%print("")
301 call die_msg(698012538, "Property type mismatch for "//key// &
302 " in property set of "//owner_name)
303 end select
304 class default
305 call new_link%print("")
306 call die_msg(359604264, "Trying to overwrite property "//key// &
307 " in property set of "//owner_name)
308 end select
309 return
310 end if
311
312 end if
313
314 end if
315
316 ! create a new link. for property_t sub-property sets,
317 ! copy the passed value to a new object
318 select type (val)
319 class is (property_t)
320 allocate(sub_prop_set)
321 sub_link => val%first_link
322 do while (associated(sub_link))
323 call sub_prop_set%put(sub_link%key_name, sub_link%val, &
324 allow_duplicates, owner_name)
325 sub_link => sub_link%next_link
326 end do
327 new_link => property_link_t(key, sub_prop_set)
328 sub_prop_set%first_link => null()
329 sub_prop_set%last_link => null()
330 deallocate(sub_prop_set)
331 class default
332 new_link => property_link_t(key, val)
333 end select
334
335 ! if the key does not exist in the property dataset,
336 ! create a new link to add it.
337 if (.not.associated(this%first_link)) then
338 this%first_link => new_link
339 this%last_link => this%first_link
340 else
341 this%last_link%next_link => new_link
342 this%last_link => new_link
343 end if
344
345 this%num_elem = this%num_elem + 1
346
347 end subroutine put

733 !> Set the value of a property key-value pair
734 subroutine set_value(this, val)
735
736 !> Property key-value pair
737 class(property_link_t), intent(inout) :: this
738 !> New value
739 class(*), intent(in) :: val
740
741 type(string_t), pointer :: str_val
742
743 ! determine the value type
744 select type(val)
745
746 ! add integers, reals, logicals, and string_t as-is
747 type is (integer(kind=i_kind))
748 type is (real(kind=dp))
749 type is (logical)
750 type is (string_t)
751
752 ! handle empty sub-sets
753 class is (property_t)
754 if (.not.associated(val%first_link)) then
755 this%val => property_t()
756 return
757 end if
758
759 ! convert character arrays to string_t objects
760 type is (character(len=*))
761 allocate(str_val)
762 str_val%string = val
763 this%val => str_val
764 return
765
766 ! error on unsupported types
767 class default
768 call die_msg(728532218, "Unsupported property type")
769 end select
770
771 allocate(this%val, source=val)
772
773 end subroutine set_value
